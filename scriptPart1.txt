    function generateCerts() {
     which cryptogen
     if [ "$?" -ne 0 ]; then
     echo "cryptogen tool not found. exiting"
     exit 1
     fi
    
     if [ -d "crypto-config" ]; then
     rm -Rf crypto-config
     fi
     set -x
     cryptogen generate --config=./crypto-config.yaml
     res=$?
     set +x
     if [ $res -ne 0 ]; then
     echo "Failed to generate certificates..."
     exit 1
     fi
    }
